sql-- Check: how many of T8VDV's "current" rows have a NULL in the key?
SELECT
    SUM(CASE WHEN CO_ID IS NULL THEN 1 ELSE 0 END) AS NULL_CO_ID,
    SUM(CASE WHEN POL_ID IS NULL THEN 1 ELSE 0 END) AS NULL_POL_ID,
    SUM(CASE WHEN CVG_NUM IS NULL THEN 1 ELSE 0 END) AS NULL_CVG_NUM,
    SUM(CASE WHEN CVG_DIV_DT IS NULL THEN 1 ELSE 0 END) AS NULL_CVG_DIV_DT,
    COUNT(*) AS TOTAL_CURRENT_ROWS
FROM KCDWHPRC.T8VDV
WHERE CRNT_REC_IND = 'Y';
sql-- Same check for T9DVP
