SELECT
    IX.INDNAME,
    IX.UNIQUERULE,
    IXC.COLNAME,
    IXC.COLSEQ
FROM SYSCAT.INDEXES IX
JOIN SYSCAT.INDEXCOLUSE IXC
    ON IX.INDSCHEMA = IXC.INDSCHEMA
    AND IX.INDNAME = IXC.INDNAME
WHERE IX.TABSCHEMA = 'KCDWHPRC'
  AND IX.TABNAME = 'T8LPUA'
ORDER BY IX.INDNAME, IXC.COLSEQ;

-- For NON-flagged keys (i.e., excluding our known 19 broken ones),
-- how often does one CO_ID/POL_ID/CVG_NUM/DIV_EFF_YR/L7FD_PECD combo
-- have MULTIPLE DISTINCT DIV_EFF_DT values across its full history?
SELECT
    CO_ID, POL_ID, CVG_NUM, YEAR(DIV_EFF_DT) AS DIV_EFF_YR, L7FD_PECD,
    COUNT(DISTINCT DIV_EFF_DT) AS DISTINCT_DATES_IN_YEAR,
    COUNT(DISTINCT CVG_DIV_OPT_CD) AS DISTINCT_OPT_CODES_IN_YEAR,
    COUNT(*) AS TOTAL_ROWS
FROM KCDWHPRC.T8LPUA
GROUP BY CO_ID, POL_ID, CVG_NUM, YEAR(DIV_EFF_DT), L7FD_PECD
HAVING COUNT(DISTINCT DIV_EFF_DT) > 1
ORDER BY DISTINCT_DATES_IN_YEAR DESC;
