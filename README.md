SELECT
    F.CO_ID, F.POL_ID, F.CVG_NUM, YEAR(F.DIV_EFF_DT) AS DIV_EFF_YR, F.L7FD_PECD,
    F.RPT_PRCES_STATS_ID,
    COUNT(*) AS ROWS_IN_SAME_RUN,
    COUNT(DISTINCT F.DIV_EFF_DT) AS DISTINCT_DATES_IN_RUN,
    COUNT(DISTINCT F.CVG_DIV_OPT_CD) AS DISTINCT_OPT_CODES_IN_RUN
FROM KCSTGPRC.L70_FTRLR F
WHERE F.CRNT_REC_IND = 'Y'
GROUP BY F.CO_ID, F.POL_ID, F.CVG_NUM, YEAR(F.DIV_EFF_DT), F.L7FD_PECD, F.RPT_PRCES_STATS_ID
HAVING COUNT(*) > 1
ORDER BY ROWS_IN_SAME_RUN DESC;
