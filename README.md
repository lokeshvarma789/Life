-- Exact count for each table, one clean summary instead of "more than 50"
SELECT 'T8VDV' AS TABLE_NAME, COUNT(*) AS DUPLICATE_KEY_COUNT
FROM (
    SELECT CO_ID, POL_ID, CVG_NUM, CVG_DIV_DT
    FROM KCDWHPRC.T8VDV
    WHERE CRNT_REC_IND = 'Y'
    GROUP BY CO_ID, POL_ID, CVG_NUM, CVG_DIV_DT
    HAVING COUNT(*) > 1
)
UNION ALL
SELECT 'T9DVP', COUNT(*)
FROM (
    SELECT CO_ID, POL_ID, REC_TYP_CD
    FROM KCDWHPRC.T9DVP
    WHERE CRNT_REC_IND = 'Y' AND DERIV_VALU_FREQ_CD = 'C'
    GROUP BY CO_ID, POL_ID, REC_TYP_CD
    HAVING COUNT(*) > 1
)
UNION ALL
SELECT 'T8LPOLL', COUNT(*)
FROM (
    SELECT CO_ID, POL_ID
    FROM KCDWHPRC.T8LPOLL
    WHERE CRNT_REC_IND = 'Y' AND CO_ID IN ('01','50') AND POL_ID > '0'
    GROUP BY CO_ID, POL_ID
    HAVING COUNT(*) > 1
)
UNION ALL
SELECT 'T8LPUA', COUNT(*)
FROM (
    SELECT SOURCE, CO_ID, POL_ID, CVG_NUM, DIV_EFF_DT, CVG_DIV_OPT_CD, L7FD_PECD
    FROM KCDWHPRC.T8LPUA
    WHERE CRNT_REC_IND = 'Y'
    GROUP BY SOURCE, CO_ID, POL_ID, CVG_NUM, DIV_EFF_DT, CVG_DIV_OPT_CD, L7FD_PECD
    HAVING COUNT(*) > 1
);


-- Same test we ran on T8LPUA - do the duplicate pairs come from ONE
-- run colliding with itself, or TWO separate runs (one failing to
-- see the other)?
WITH FLAGGED AS (
    SELECT CO_ID, POL_ID, CVG_NUM, CVG_DIV_DT
    FROM KCDWHPRC.T8VDV
    WHERE CRNT_REC_IND = 'Y'
    GROUP BY CO_ID, POL_ID, CVG_NUM, CVG_DIV_DT
    HAVING COUNT(*) > 1
)
SELECT T.CO_ID, T.POL_ID, T.CVG_NUM, T.CVG_DIV_DT,
       COUNT(*) AS DUP_ROWS,
       COUNT(DISTINCT T.RPT_PRCES_STATS_ID) AS DISTINCT_RUNS_INVOLVED
FROM KCDWHPRC.T8VDV T
INNER JOIN FLAGGED F
    ON T.CO_ID = F.CO_ID AND T.POL_ID = F.POL_ID
    AND T.CVG_NUM = F.CVG_NUM AND T.CVG_DIV_DT = F.CVG_DIV_DT
WHERE T.CRNT_REC_IND = 'Y'
GROUP BY T.CO_ID, T.POL_ID, T.CVG_NUM, T.CVG_DIV_DT
ORDER BY DISTINCT_RUNS_INVOLVED DESC;
